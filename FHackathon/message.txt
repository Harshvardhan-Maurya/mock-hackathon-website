    <script>
        // Auto-detect region (India)
        document.getElementById("user-region").textContent = "India";
        document.getElementById("focus-region").textContent = "India";
        document.getElementById("why-region").textContent = "India";
        document.getElementById("why-region-text").textContent = "India";

        // Set the countdown to 22nd September 2025, 11:30 PM IST
        const countDownDate = new Date("September 22, 2025 23:30:00").getTime();

        let x = setInterval(function() {
            const now = new Date().getTime();
            const distance = countDownDate - now;

            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById("countdown-hours").innerHTML = String(hours).padStart(2, '0');
            document.getElementById("countdown-minutes").innerHTML = String(minutes).padStart(2, '0');
            document.getElementById("countdown-seconds").innerHTML = String(seconds).padStart(2, '0');

            if (distance < 0) {
                clearInterval(x);
                document.getElementById("countdown").innerHTML = "<p class='text-red-500 font-bold'>The Hackathon has ended!</p>";
            }
        }, 1000);
    </script>
